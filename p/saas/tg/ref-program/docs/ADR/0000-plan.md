## 1. Архитектурный обзор

### 1.1 Назначение системы

Партнёрская программа — веб-приложение для трекинга привлечения клиентов через партнёрские ссылки с интеграцией Telegram Mini App. Система фиксирует путь клиента от клика по ссылке до оплаты.

### 1.2 Основные сущности

| Сущность | Описание | Хранение |
|----------|----------|----------|
| **Пользователь (User)** | Администратор панели, авторизуется через веб | Системная таблица Chatium |
| **Кампания (Campaign)** | Изолированный контейнер данных | Heap: `campaigns` |
| **Партнёр (Partner)** | Telegram-пользователь, привлекает трафик | Heap: `partners` |
| **Реферал (Referral)** | Конечный клиент | Heap: `referrals` |
| **Визит (Visit)** | Клик по ссылке | Heap: `visits` |
| **Событие** | Регистрация/Заказ/Оплата/Custom | Heap: `registrations`, `orders`, `payments` |

### 1.3 Технологический стек

- **Backend**: Chatium V8 Runtime, TypeScript
- **Frontend**: Vue 3.5.13 (Composition API, SSR)
- **База данных**: Heap (встроенная БД Chatium)
- **Стили**: TailwindCSS 3.4.16
- **Иконки**: FontAwesome 6.7.2
- **Мессенджер**: Telegram Bot API + Mini Apps

### 1.4 Роли и доступы

```
Системные роли (Chatium):
├── Admin — полный доступ ко всем кампаниям
└── User — может создавать кампании

Роли в кампании:
├── campaign-owner — владелец, полные права
└── campaign-member — участник, ограниченные права
```

async function handleRegisterWebhook(
  ctx: app.Ctx,
  data: Record<string, unknown>
): Promise<{ success: boolean; error?: string }> {
  const { key, ref, tg_id, gc_id, name, email, phone, ...rest } = data
  
  // Логируем входящий запрос
  ctx.account.log('Register webhook received', {
    level: 'info',
    json: { key, ref, tg_id, gc_id, name, hasEmail: !!email, hasPhone: !!phone }
  })
  
  // Валидация обязательных параметров
  if (!key || typeof key !== 'string') {
    return { success: false, error: 'Missing key parameter' }
  }
  
  if (!ref || typeof ref !== 'string') {

  }

  const campaign = await campaignRepo.findCampaignBySecret(ctx, key)
  if (!campaign) {
    ctx.account.log('Campaign not found by secret', {
      level: 'warn',
      json: { key }
    })
    return { success: false, error: 'Invalid key' }
  }
  
  // Обрабатываем регистрацию
  try {
    const result = await eventRepo.processRegistration(ctx, campaign.id, {
      ref: ref as string,
      tgId: tg_id as string,
      gcId: gc_id as string,
      name: name as string,
      email: email as string,
      phone: phone as string,
      rawPayload: { ...data, receivedAt: new Date().toISOString() }
    })
    
    ctx.account.log('Registration processed', {
      level: 'info',
      json: {
        campaignId: campaign.id,
        ref,
        isNew: result.isNew
      }
    })
    
    return { success: true }
  } catch (error: any) {
    ctx.account.log('Registration processing error', {
      level: 'error',
      json: {
        campaignId: campaign.id,
        ref,
        error: error.message
      }
    })
    
    return { success: false, error: 'Processing error' }
  }
}

### 9.1 lib/core/fingerprint.ts

**Назначение**: Генерация fingerprint для дедупликации визитов.

```typescript
/**
 * Данные для fingerprint
 */
export interface FingerprintData {
  ip: string
  userAgent: string
  acceptLanguage: string
  platform?: string
  timezone?: string
}

/**
 * Результат вычисления fingerprint
 */
export interface FingerprintResult {
  hash: string
  parts: FingerprintData
}

/**
 * Вычисление fingerprint из данных запроса
 */
export function computeFingerprint(req: app.Req): FingerprintResult {
  const parts: FingerprintData = {
    ip: getClientIp(req),
    userAgent: req.headers['user-agent'] || '',
    acceptLanguage: req.headers['accept-language'] || '',
    platform: req.headers['sec-ch-ua-platform'] || '',
    timezone: req.headers['sec-ch-timezone'] || ''
  }
  
  // Вычисляем хеш
  // Примечание: crypto недоступна в Chatium, используем простой хеш
  const hash = simpleHash(JSON.stringify(parts))
  
  return { hash, parts }
}

/**
 * Получение IP клиента
 */
function getClientIp(req: app.Req): string {
  // Проверяем X-Forwarded-For
  const forwarded = req.headers['x-forwarded-for']
  if (forwarded) {
    const ips = forwarded.split(',')
    return ips[0].trim()
  }
  
  // Fallback
  return req.headers['x-real-ip'] || 'unknown'
}

/**
 * Простая хеш-функция (замена crypto)
 */
function simpleHash(str: string): string {
  let hash = 0
  for (let i = 0; i < str.length; i++) {
    const char = str.charCodeAt(i)
    hash = ((hash << 5) - hash) + char
    hash = hash & hash // Convert to 32bit integer
  }
  return Math.abs(hash).toString(36)
}
```

/**
 * Алфавит для URL-safe идентификаторов (base62)
 */
const ALPHABET = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789'

/**
 * Генерация URL-safe идентификатора
 * @param length Длина идентификатора (по умолчанию 8)
 */
export function generateUrlSafeId(length: number = 8): string {
  let result = ''
  
  for (let i = 0; i < length; i++) {
    const randomIndex = Math.floor(Math.random() * ALPHABET.length)
    result += ALPHABET[randomIndex]
  }
  
  return result
}

/**
 * Генерация реферального идентификатора
 * Гарантированно уникальный в рамках кампании
 */
export async function generateUniqueRef(
  ctx: app.Ctx,
  campaignId: string,
  maxAttempts: number = 10
): Promise<string> {
  for (let i = 0; i < maxAttempts; i++) {
    const ref = generateUrlSafeId(8)
    
    // Проверяем уникальность
    const existing = await VisitsTable.findOneBy(ctx, {
      campaignId,
      ref
    })
    
    if (!existing) {
      return ref
    }
  }
  
  throw new Error('Failed to generate unique ref after max attempts')
}

/**
 * Запуск всех тестов
 */
export async function runAllTests(ctx: app.Ctx): Promise<{
  total: number
  passed: number
  failed: number
  results: TestResult[]
}> {
  const results: TestResult[] = []

  for (const test of allTests) {
    const result = await test.run(ctx)
    results.push(result)
  }

  return {
    total: results.length,
    passed: results.filter(r => r.success).length,
    failed: results.filter(r => !r.success).length,
    results
  }
}

/**
 * Запуск тестов по категории
 */
export async function runTestsByCategory(
  ctx: app.Ctx,
  category: 'unit' | 'integration' | 'e2e'
): Promise<{
  total: number
  passed: number
  failed: number
  results: TestResult[]
}> {
  const tests = allTests.filter(t => t.category === category)
  const results: TestResult[] = []

  for (const test of tests) {
    const result = await test.run(ctx)
    results.push(result)
  }

  return {
    total: results.length,
    passed: results.filter(r => r.success).length,
    failed: results.filter(r => !r.success).length,
    results
  }
}
